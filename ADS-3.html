                                        function getNumbers(number) {
    let arr = [];

    for (let i = 1; i <= number; i++) {
        let k = 0;
        let l = 0;
        let m = 0;
        let p = i;

        while (p % 3 == 0) {
            k++;
            p /= 3;
        }

        while (p % 5 == 0) {
            l++;
            p /= 5;
        }

        while (p % 7 == 0) {
            m++;
            p /= 7;
        }

        if (p == 1) {
            arr.push(i);
        }
    }

    return arr;
}
