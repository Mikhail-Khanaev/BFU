Методы сортировки. Сортировка вставками

Дана последовательность чисел. Отсортировать и вывести последовательность чисел, определённым методом

Принцип алгоритма

Смотрим элементы массива от 2 до последнего если число на позиции pos-1 больше числа на позиции pos двигаем число на позиции pos влево, пока число перед ним не будет больше либо пока число не встанет на 1 место массива

                                        nums = list(map(int, input("Введите числовую последовательность\n").split(" ")))
for i in range(1, len(nums)):
    if nums[i] < nums[i-1]:
        pos = i
        while pos > 0:
            if nums[pos] < nums[pos-1]:
                nums[pos], nums[pos - 1] = nums[pos - 1], nums[pos]
                pos -= 1
            else:
                break
print("Отсортированная последовательность")
print(*nums)

#include <vector>
#include <limits>

using namespace std;

vector<int> input() {
  vector<int> storage;
  int n;

  cin >> n;

  for (int i = 0; i < n; i++) {
    int temp;
    cin >> temp;
    storage.push_back(temp);
  }
  return storage;
}

void printer(vector<int> arr) {
  for (int i = 0; i < arr.size(); i++) {
    cout << arr[i] << " ";
  }
  cout << endl;
}


void lab5() {
  vector<int> storage = input();

  for (int i = 1; i < storage.size(); i++) {
    if (storage[i] < storage[i - 1]) {
      int pos = i;
      while (pos > 0) {
        if (storage[pos] < storage[pos - 1]) {
          int buffer = storage[pos];
          storage[pos] = storage[pos - 1];
          storage[pos - 1] = buffer;
          pos--;
        }
        else {
          break;
        }
      }
    }
  }
  printer(storage);
}


int main() {
  lab5();
  return 0;
}
