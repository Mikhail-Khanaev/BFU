                                        function checkBrackets(str) {
    let stack = [];
    let obj = {
        "{" : "}",
        "(" : ")",
        "[" : "]",
    }

    for (let i = 0; i < str.length; i++) {
        const bracket = str[i];

        if (bracket === "(" || bracket === "{" || bracket === "[") {
            stack.push(bracket);
        } else {
            const lastElement = stack.pop();

            if (bracket !== obj[lastElement]) return false;
        }
    }

    if (stack.length !== 0) return false;
    return true;
}
