                                        const hashTableSize = 20;

class HashTable {
    constructor() {
        this.array = Array(hashTableSize).fill(null);
    }

    hash(key) {
        let sum = 0;

        for (let i = 0; i < key.length; i++) {
            sum += key.charCodeAt(i);
        }

        return sum % hashTableSize;
    }

    set(value) {
        let key = this.hash(value);

        if (this.array[key] == null) {
            this.array[key] = [];
        }

        let list = this.array[key];
        list.push(value);
    }
}
