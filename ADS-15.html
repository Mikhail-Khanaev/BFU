Методы сортировки. Сортировка посредством выбора

Дана последовательность чисел. Отсортировать и вывести последовательность чисел, определённым методом

Принцип алгоритма

Суть алгоритма — за каждый проход по массиву выбрать минимальный элемент (для сортировки по возрастанию) и поменять его местами с первым элементом в еще не отсортированном участке массива, тем самым уменьшив длину этого участка на один, и так до тех пор пока не будут отсортированы все элементы.

Этот подход хорошо виден на иллюстрации:

                                                                                        // Сортировка посредством выбора
function SortingBySelection(array) {
    let arrayLength = array.length;
    let k = arrayLength - 1;
    for (let i = 0; i < k; i++) {
        let indexMin = i;
        for (let j = i + 1; j < arrayLength; j++) {
            if (array[indexMin] > array[j]) {
                indexMin = j;
            }
        }
        if (indexMin !== i) {
            [array[i], array[indexMin]] = [array[indexMin], array[i]];
        }
    }
    return array;
}

                                        nums = list(map(int, input("Введите числовую последовательность\n").split(" ")))
MIN_VALUE = 2**15 + 1

for i in range(len(nums)):
    min_value, pos = MIN_VALUE, 0
    for j in range(i, len(nums)):
        if nums[j] < min_value:
            min_value = nums[j]
            pos = j
    nums[pos], nums[i] = nums[i], nums[pos]
print("Отсортированная последовательность")
print(*nums)

void printer(vector<int> arr) {
  for (int i = 0; i < arr.size(); i++) {
    cout << arr[i] << " ";
  }
  cout << endl;
}


void lab6() {
  vector<int> storage = input();
  int min_value, pos;
  for (int i = 0; i < storage.size(); i++) {
    min_value = numeric_limits<int>::max();
    pos = 0;
    for (int j = i; j < storage.size(); j++) {
      if (storage[j] < min_value) {
        min_value = storage[j];
        pos = j;
      }
    }
    int buffer = storage[pos];
    storage[pos] = storage[i];
    storage[i] = buffer;
  }
  printer(storage);
}
