Рекурсивные обходы (прямой, центральный, кольцевой)

В качестве входной информации для следующих лабораторных работ вводится символьная строка (бинарное дерево) с помощью линейно-скобочной записи. Например 8 (3 (1, 6 (4, 7), 10 (, 14 (13,))))

                                        // Класс для создания узлов в дереве
class Node {
    constructor(value) {
        this.value = value;
        this.left = null;
        this.right = null;
    }
}

class BinaryThree {
    constructor() {
        this.root = null; // По умолчанию корень дерева равен null
    }

    // Метод добавления элементов в дерево
    add(value) {
        let newNode = new Node(value);

        // Если нет корня, то добавляем его 
        if (!this.root) {
            this.root = newNode;
            return;
        }

        // Проходим с помощью цикла по всем элементам дерева
        let currentNode = this.root; // Начинаем с корневого элемента
        while (currentNode) {
            // Если новый элемент меньше, то помещаем его влево
            if (newNode.value < currentNode.value) {
                if (!currentNode.left) {
                    currentNode.left = newNode;
                    return;
                }

                // но если там уже "занято", 
                // то спускаемся на один уровень вниз
                currentNode = currentNode.left;
            } else {
                // Если новый элемент больше, то помещаем его впрово
                if (!currentNode.right) {
                    currentNode.right = newNode;
                    return;
                }

                // но если там уже "занято",
                // то спускаемся на один уровень вниз
                currentNode = currentNode.right;
            }
        }
    }

    // Прямой обход
    // Сначала корневой узел
    // Потом все узлы левого поддереве
    // Потом все узлы правого поддрева
    preOrder(node, callback) {

        if (!node) {
            return;
        }

        // Сначала корневой узел
        if (callback) {
            callback(node);
        }

        // Потом все узлы левого поддереве
        this.preOrder(node.left, callback);
        // Потом все узлы правого поддрева
        this.preOrder(node.right, callback);

    }

    // Центральный обход
    // Сначала все узлы левого поддерева
    // Потом корневой узел
    // Потом все узлы правого поддрева
    inOrder(node, callback) {
        if (!node) {
            return;
        }

        // Сначала все узлы левого поддереве
        this.inOrder(node.left, callback);

        // Потом корневой узел
        if (callback) {
            callback(node);
        }

        // Потом все узлы правого поддрева
        this.inOrder(node.right, callback);
    }

    // Обратный обход (или же концевой)
    // Сначала все узлы левого поддерева
    // Потом все узлы правого поддрева
    // Потом корневой узел
    postOrder(node, callback) {
        if (!node) {
            return;
        }

        // Сначала все узлы левого поддереве
        this.postOrder(node.left, callback);

        // Потом все узлы правого поддрева
        this.postOrder(node.right, callback);

        // Потом корневой узел
        if (callback) {
            callback(node);
        }
    }


    // Обход дерева в глубину (от корня)
    // Depth Firts Search
    traverseDFS(callback, method) {
        if (method === 'preOrder') {
            return this.preOrder(this.root, callback);
        }
        if (method === 'inOrder') {
            return this.inOrder(this.root, callback);
        }

        return this.postOrder(this.root, callback);
    }

    // Обход дерева в ширину (по уровням)
    // Или же нерекурсивный прямой обход 
    // Breadth First Seacrh
    // Я буду использовать структуру данных "очередь"
    // Первым пришёл - первым ушёл 
    // First in first out (FIFO)
    traverseBFS(callback) {
        // делаем очередь, с первым корневым элементом
        // (так как он, очевидно, на 1 уровне)
        const queue = [this.root];

        // пока очередь существует
        while (queue.length) {
            let node = queue.shift(); // Достаем элемент из очереди
            callback(node) // Печатаем наш элемент

            // если у элемента был левый узел, 
            // то добавляем его в очередь
            if (node.left) {
                queue.push(node.left);
            }

            // Аналогично с правым
            if (node.right) {
                queue.push(node.right);
            }
        }
    }
}
