                                        // Сортировка Слиянием
function MergeSort(array) {
    // Вспомогательная функция, 
    // которая и отвечает за слияние двух подмассивов
    let merge = (firstArray, secondArray) => {
        let sorterdArray = [];
        let i = 0;
        let j = 0;

        // сравниваем два массива, поочередно сдвигая указатели
        // пока оба указателя находятся в своём диапазоне, 
        // будем поочередно их сдвигать, добавляя в итоговый массив наименьший элемент
        while (i < firstArray.length && j < secondArray.length) {

            if (firstArray[i] < secondArray[j]) {
                sorterdArray.push(firstArray[i]);
                i++;
            } else {
                sorterdArray.push(secondArray[j]);
                j++;
            }
        }


        /* Поскольку цикл заканчивается тогда,
        когда один из указателей доходит до конца, 
        то может получиться так,
        что некоторые элементы не попадут в итоговый массив.
        Чтобы этого избежать, нужно добавить два дублирующих цикла, 
        которые как бы перепроверяют работу первого */
        while (i < firstArray.length) {
            sorterdArray.push(firstArray[i]);
            i++;
        }

        while (j < secondArray.length) {
            sorterdArray.push(secondArray[j]);
            j++;
        }

        // Возвращаем отсортированный массив
        return sorterdArray;
    }

    // Если длина списка равна 1, 
    // то мы выходим из рекурсии и  Возвращаем массив,
    // (поскольку там один элемент, он уже отсортирован)
    if (array.length <= 1) {
        return array;
    }

    // считаем серидину и делаем левый и правый срез, чтобы получить два массива
    let middle = Math.floor(array.length / 2);
    let arrayLeft = array.slice(0, middle);
    let arrayRight = array.slice(middle);

    // теперь рекурсивно вызваем функцию слияния
    return merge(MergeSort(arrayLeft), MergeSort(arrayRight))
}
